### Next.js Image Component

**CardImage Implementation:**

```typescript
// components/cards/CardImage.tsx
'use client';

import Image from 'next/image';
import { Card } from '@prisma/client';
import { useState } from 'react';

interface CardImageProps {
  card: Card;
  size?: 'sm' | 'md' | 'lg';
  priority?: boolean;
  className?: string;
}

const SIZE_MAP = {
  sm: { width: 200, height: 320 },
  md: { width: 400, height: 640 },
  lg: { width: 800, height: 1280 },
};

export function CardImage({
  card,
  size = 'md',
  priority = false,
  className = ''
}: CardImageProps) {
  const [imgSrc, setImgSrc] = useState(card.image_url);
  const [isLoading, setIsLoading] = useState(true);

  const dimensions = SIZE_MAP[size];
  const altText = `ไพ่ยิปซี ${card.name_th}`;

  return (
    <div className={`relative ${className}`}>
      {isLoading && (
        <div className="absolute inset-0 bg-slate-700 animate-pulse rounded-lg" />
      )}
      <Image
        src={imgSrc}
        alt={altText}
        width={dimensions.width}
        height={dimensions.height}
        priority={priority}
        loading={priority ? 'eager' : 'lazy'}
        sizes="(max-width: 768px) 100vw, (max-width: 1200px) 50vw, 33vw"
        onLoadingComplete={() => setIsLoading(false)}
        onError={() => {
          setImgSrc('/cards/placeholder.webp');
          setIsLoading(false);
        }}
        className="rounded-lg shadow-lg"
      />
    </div>
  );
}
```

### Card Flip Animation

**3D Flip Implementation:**

```typescript
// components/cards/CardFlip.tsx
'use client';

import { motion } from 'framer-motion';
import { Card } from '@prisma/client';
import { CardImage } from './CardImage';

interface CardFlipProps {
  card: Card;
  isRevealed: boolean;
  onFlipComplete?: () => void;
  size?: 'sm' | 'md' | 'lg';
}

export function CardFlip({
  card,
  isRevealed,
  onFlipComplete,
  size = 'md'
}: CardFlipProps) {
  return (
    <div className="perspective-1000">
      <motion.div
        className="relative preserve-3d"
        animate={{ rotateY: isRevealed ? 180 : 0 }}
        transition={{
          duration: 0.8,
          ease: 'easeInOut'
        }}
        onAnimationComplete={onFlipComplete}
        style={{ transformStyle: 'preserve-3d' }}
      >
        {/* Card Back */}
        <div className="absolute backface-hidden">
          <Image
            src="/cards/back.webp"
            alt="Card back"
            width={400}
            height={640}
            className="rounded-lg"
          />
        </div>

        {/* Card Front */}
        <div
          className="backface-hidden"
          style={{ transform: 'rotateY(180deg)' }}
        >
          <CardImage card={card} size={size} />
        </div>
      </motion.div>
    </div>
  );
}
```

**Required CSS (globals.css):**

```css

  perspective: 1000px;
}

  transform-style: preserve-3d;
}

  backface-visibility: hidden;
}
```

### Image Optimization Script

**Sharp-based Optimization:**

```typescript
// scripts/optimize-images.ts
import sharp from 'sharp';
import { readdirSync, mkdirSync } from 'fs';
import { join } from 'path';

const SIZES = {
  sm: 200,
  md: 400,
  lg: 800,
};

const inputDir = 'assets/cards/original';
const outputDir = 'public/cards';

async function optimizeImage(inputPath: string, outputPath: string, width: number) {
  await sharp(inputPath)
    .resize(width, Math.floor(width * 1.6)) // 5:8 aspect ratio
    .webp({ quality: 82 })
    .toFile(outputPath);

  const stats = await sharp(outputPath).metadata();
  console.log(`✓ ${outputPath} (${Math.floor(stats.size / 1024)}KB)`);
}

async function main() {
  // Process all card images
  const suits = ['major-arcana', 'wands', 'cups', 'swords', 'pentacles'];

  for (const suit of suits) {
    const files = readdirSync(join(inputDir, suit));

    for (const file of files) {
      const inputPath = join(inputDir, suit, file);
      const fileName = file.replace(/\.(jpg|png)$/, '.webp');

      // Generate all 3 sizes
      for (const [sizeName, width] of Object.entries(SIZES)) {
        const outputPath = join(outputDir, suit, fileName);
        await optimizeImage(inputPath, outputPath, width);
      }
    }
  }

  console.log('✓ All images optimized!');
}

main();
```

### Accessibility

**Alt Text Best Practices:**

- Descriptive: "The Fool tarot card" not just "Card"
- Include Thai name: "ไพ่ยิปซีคนบ้า"
- Mention reversed if applicable: "ไพ่ยิปซีคนบ้า (กลับหัว)"
- Don't include "image of" (redundant for screen readers)

**Image Loading UX:**

- Show skeleton/blur placeholder while loading
- Maintain aspect ratio to prevent layout shift (CLS)
- Use `priority` for above-fold images
- Lazy load below-fold images
